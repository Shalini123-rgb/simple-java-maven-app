pipeline {
    agent any

    tools {
        maven 'mvn'
    }

    stages {
        stage("Fetching") {
            steps {
                git branch: "master", 
                url: "https://github.com/techwithapd/artigo-boas-praticas-medium"
            }
        }
        stage("Compile") {
            steps {
                bat 'mvn compile'
            }
        }
        stage("Test"){
            steps{
                bat 'mvn test'
            }
        }
        stage("Package"){
            steps{
                bat 'mvn package'
            }
        }
        stage("Verify"){
            steps{
                bat 'mvn verify'
            }
        }
        stage("Install"){
            steps{
                bat 'mvn install'
            }
        }
    }
}
